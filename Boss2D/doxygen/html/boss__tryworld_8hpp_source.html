547  private: const Points* MERGE_POLYGON(const Points& Dst, const Points& Src, const Rect SrcBound, const bool IsBoundLine)
548  {
549  // 한계검사
550  bool IsL = true, IsT = true, IsR = true, IsB = true;
551  int BestDstI = -1;
552  for(int i = 0; i < Dst.Count(); ++i)
553  {
554  if(IsL) IsL &= (Dst[i].x < SrcBound.l);
555  if(IsT) IsT &= (Dst[i].y < SrcBound.t);
556  if(IsR) IsR &= (Dst[i].x > SrcBound.r);
557  if(IsB) IsB &= (Dst[i].y > SrcBound.b);
558  if(!IsBoundLine && BestDstI == -1 && (Dst[i].x < SrcBound.l || Dst[i].y < SrcBound.t))
559  BestDstI = i;
560  }
561  BOSS_ASSERT("Src는 Dst의 우측하단에 위치하여야 안전합니다", IsBoundLine || BestDstI != -1);
562  if(IsL || IsT || IsR || IsB) return nullptr;
563 
564  // 교차검사 수집데이터(최소한 Dst의 0번 정점은 Src에 포함되지 않아야 함)
565  static Points CollectDstB;
566  static Points CollectDstE;
567  int CollectDstLength = Dst.Count();
568  Memory::Copy(CollectDstB.AtDumping(0, CollectDstLength), &Dst[0], sizeof(Point) * CollectDstLength);
569  Memory::Copy(CollectDstE.AtDumping(0, CollectDstLength), &Dst[1], sizeof(Point) * (CollectDstLength - 1));
570  CollectDstE.At(CollectDstLength - 1) = Dst[0];
571  static Points CollectSrcB;
572  static Points CollectSrcE;
573  int CollectSrcLength = Src.Count();
574  Memory::Copy(CollectSrcB.AtDumping(0, CollectSrcLength), &Src[0], sizeof(Point) * CollectSrcLength);
575  Memory::Copy(CollectSrcE.AtDumping(0, CollectSrcLength), &Src[1], sizeof(Point) * (CollectSrcLength - 1));
576  CollectSrcE.At(CollectSrcLength - 1) = Src[0];
577  for(int d = 0; d < CollectDstLength; ++d)
578  for(int s = 0; s < CollectSrcLength; ++s)
579  {
580  const Point* CrossDot = nullptr;
581  if(CrossDot = Util::GetDotByLineCross(CollectDstB[d], CollectDstE[d], CollectSrcB[s], CollectSrcE[s]))
582  {
583  if((CrossDot->x != CollectDstB[d].x || CrossDot->y != CollectDstB[d].y)
584  && (CrossDot->x != CollectDstE[d].x || CrossDot->y != CollectDstE[d].y))
585  {
586  CollectDstE.AtWherever(CollectDstLength) = CollectDstE[d];
587  CollectDstB.AtWherever(CollectDstLength++) = (CollectDstE.At(d) = *CrossDot);
588  }
589  if((CrossDot->x != CollectSrcB[s].x || CrossDot->y != CollectSrcB[s].y)
590  && (CrossDot->x != CollectSrcE[s].x || CrossDot->y != CollectSrcE[s].y))
591  {
592  CollectSrcE.AtWherever(CollectSrcLength) = CollectSrcE[s];
593  CollectSrcB.AtWherever(CollectSrcLength++) = (CollectSrcE.At(s) = *CrossDot);
594  }
595  }
596  }
597  if(CollectDstLength == Dst.Count())
598  return nullptr;
599 
600  static Points Result;
601  Result.SubtractionAll();
602  if(IsBoundLine)
603  {
604  Result.AtAdding() = Point(0, 0);
605  for(int d = 0; d < CollectDstLength; ++d)
606  {
607  if(CollectDstB[d].x <= SrcBound.l) continue;
608  if(CollectDstB[d].y <= SrcBound.t) continue;
609  if(CollectDstB[d].x >= SrcBound.r) continue;
610  if(CollectDstB[d].y >= SrcBound.b) continue;
611  Result.At(0) = CollectDstB[d];
612  break;
613  }
614  BOSS_ASSERT("계산상 오류입니다", Result[0].x != 0 || Result[0].y != 0);
615  }
616  else Result.At(0) = Dst[BestDstI];
617  do
618  {
619  int DstIndex = -1;
620  for(int d = 0; d < CollectDstLength; ++d)
621  if(Result[-1].x == CollectDstB[d].x && Result[-1].y == CollectDstB[d].y
622  && (CollectDstB[d].x != CollectDstE[d].x || CollectDstB[d].y != CollectDstE[d].y))
623  {
624  DstIndex = d;
625  break;
626  }
627  int SrcIndex = -1;
628  for(int s = 0; s < CollectSrcLength; ++s)
629  if(Result[-1].x == CollectSrcB[s].x && Result[-1].y == CollectSrcB[s].y
630  && (CollectSrcB[s].x != CollectSrcE[s].x || CollectSrcB[s].y != CollectSrcE[s].y))
631  {
632  SrcIndex = s;
633  break;
634  }
635  BOSS_ASSERT("계산상 오류입니다", DstIndex != -1 || SrcIndex != -1);
636  Point SelectedDot;
637  if(DstIndex != -1 && SrcIndex != -1 && 1 < Result.Count())
638  {
639  const float PrevAngle = Math::Atan(Result[-2].x - Result[-1].x, Result[-2].y - Result[-1].y);
640  const float NextAngleDst = Math::Atan(CollectDstE[DstIndex].x - Result[-1].x, CollectDstE[DstIndex].y - Result[-1].y);
641  const float NextAngleSrc = Math::Atan(CollectSrcE[SrcIndex].x - Result[-1].x, CollectSrcE[SrcIndex].y - Result[-1].y);
642  if(NextAngleDst < NextAngleSrc)
643  {
644  if(NextAngleDst < PrevAngle && PrevAngle < NextAngleSrc)
645  SelectedDot = CollectSrcE[SrcIndex];
646  else SelectedDot = CollectDstE[DstIndex];
647  }
648  else
649  {
650  if(NextAngleSrc < PrevAngle && PrevAngle < NextAngleDst)
651  SelectedDot = CollectDstE[DstIndex];
652  else SelectedDot = CollectSrcE[SrcIndex];
653  }
654  }
655  else if(DstIndex != -1)
656  SelectedDot = CollectDstE[DstIndex];
657  else SelectedDot = CollectSrcE[SrcIndex];
658  if(1 < Result.Count() && ((Result[-2].x == SelectedDot.x && Result[-1].x == SelectedDot.x)
659  || (Result[-2].y == SelectedDot.y && Result[-1].y == SelectedDot.y)))
660  Result.At(-1) = SelectedDot;
661  else Result.AtAdding() = SelectedDot;
662  }
663  while(Result[-1].x != Result[0].x || Result[-1].y != Result[0].y);
664  Result.SubtractionOne();
665  return &Result;
666  }
